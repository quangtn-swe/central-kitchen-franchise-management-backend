-- ==============================================================
-- PHÂN CÔNG: BACKEND DEV 1 (AUTH & ORDERING)
-- ==============================================================

-- 1. Bảng Phân Quyền
CREATE TABLE roles (
    role_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL UNIQUE
);

-- 2. Bảng Người Dùng
CREATE TABLE users (
    user_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    full_name VARCHAR(100),
    email VARCHAR(100),
    role_id INT REFERENCES roles(role_id),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. Bảng Cửa Hàng
CREATE TABLE stores (
    store_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    store_name VARCHAR(100) NOT NULL,
    address TEXT,
    phone VARCHAR(20),
    manager_id INT REFERENCES users(user_id)
);

-- 4. Bảng Đơn Yêu Cầu (Chỉ số lượng, không tiền)
CREATE TABLE store_orders (
    order_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    order_code VARCHAR(50) UNIQUE NOT NULL, -- 'SO-2024-001'
    store_id INT REFERENCES stores(store_id),
    order_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    delivery_date DATE,
    status VARCHAR(20) DEFAULT 'PENDING' -- 'PENDING', 'APPROVED', 'COMPLETED', 'CANCELLED'
);

-- ==============================================================
-- PHÂN CÔNG: BACKEND DEV 2 (MASTER DATA & INBOUND)
-- ==============================================================

-- 5. Bảng Danh Mục
CREATE TABLE categories (
    category_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    category_name VARCHAR(100) NOT NULL
);

-- 6. Bảng Sản Phẩm (Không giá)
CREATE TABLE products (
    product_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    product_name VARCHAR(200) NOT NULL,
    unit VARCHAR(20) NOT NULL, -- 'kg', 'lit', 'hop'
    image_url TEXT,
    description TEXT,
    category_id INT REFERENCES categories(category_id),
    status VARCHAR(20) DEFAULT 'ACTIVE'
);

-- Dev 1 cần bảng này để tạo OrderDetail, nhưng Dev 2 là người quản lý nó
CREATE TABLE order_details (
    detail_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    order_id INT REFERENCES store_orders(order_id),
    product_id INT REFERENCES products(product_id),
    quantity INT NOT NULL CHECK (quantity > 0)
);

-- 7. Bảng Lệnh Sản Xuất (Đã cập nhật quan hệ Product)
CREATE TABLE manufacturing_orders (
    manu_order_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    order_code VARCHAR(50) UNIQUE NOT NULL, -- 'MO-2024-001'
    
    -- [THÊM MỚI] Liên kết sản phẩm & Số lượng
    product_id INT REFERENCES products(product_id), 
    quantity INT NOT NULL CHECK (quantity > 0), -- Số lượng dự kiến nấu
    
    start_date TIMESTAMP WITH TIME ZONE,
    end_date TIMESTAMP WITH TIME ZONE,
    status VARCHAR(20) DEFAULT 'PLANNED', -- 'PLANNED', 'COOKING', 'COMPLETED'
    created_by INT REFERENCES users(user_id)
);


-- 8. Bảng Lô Hàng (CORE INVENTORY)
CREATE TABLE product_batches (
    batch_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    batch_code VARCHAR(50) UNIQUE NOT NULL, -- 'LOT-20240206-A'
    product_id INT REFERENCES products(product_id),
    manu_order_id INT REFERENCES manufacturing_orders(manu_order_id),
    
    initial_quantity INT NOT NULL DEFAULT 0,
    current_quantity INT NOT NULL DEFAULT 0, -- Tồn kho thực tế
    
    manufacturing_date DATE,
    expiry_date DATE, -- Quan trọng cho FEFO
    status VARCHAR(20) DEFAULT 'WAITING_FOR_STOCK' -- 'AVAILABLE', 'OUT_OF_STOCK', 'EXPIRED'
);

-- 9. Bảng Phiếu Nhập & Chi Tiết
CREATE TABLE inventory_receipts (
    receipt_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    receipt_code VARCHAR(50) UNIQUE NOT NULL, -- 'PN-001'
    receipt_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    status VARCHAR(20) DEFAULT 'COMPLETED',
    created_by INT REFERENCES users(user_id)
);

CREATE TABLE receipt_items (
    receipt_item_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    receipt_id INT REFERENCES inventory_receipts(receipt_id),
    product_batch_id INT REFERENCES product_batches(batch_id),
    quantity INT NOT NULL CHECK (quantity > 0)
);

-- ==============================================================
-- PHÂN CÔNG: BACKEND DEV 3 (OUTBOUND & AUDIT)
-- ==============================================================

-- 10. Bảng Phiếu Xuất Kho (Giao Hàng)
CREATE TABLE export_notes (
    export_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    export_code VARCHAR(50) UNIQUE NOT NULL, -- 'PX-001'
    store_order_id INT REFERENCES store_orders(order_id), -- 1 Order -> 1 Export
    export_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    status VARCHAR(20) DEFAULT 'READY', -- 'READY', 'SHIPPED'
    created_by INT REFERENCES users(user_id)
);

-- 11. Bảng Chi Tiết Xuất (Trừ từ lô nào)
CREATE TABLE export_items (
    export_item_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    export_id INT REFERENCES export_notes(export_id),
    product_batch_id INT REFERENCES product_batches(batch_id), -- Logic FEFO điền vào đây
    quantity INT NOT NULL CHECK (quantity > 0)
);

-- 12. Bảng Lịch Sử Biến Động (Sổ Cái)
CREATE TABLE inventory_transactions (
    transaction_id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    product_batch_id INT REFERENCES product_batches(batch_id),
    transaction_type VARCHAR(50) NOT NULL, -- 'IMPORT', 'EXPORT', 'DISPOSAL'
    quantity INT NOT NULL, -- +100 hoặc -50
    reference_code VARCHAR(50), -- Mã phiếu PN hoặc PX
    transaction_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    note TEXT
);

-- Indexes tối ưu hóa cho Supabase
CREATE INDEX idx_products_category ON products(category_id);
CREATE INDEX idx_batches_product_expiry ON product_batches(product_id, expiry_date);
CREATE INDEX idx_orders_store ON store_orders(store_id);
